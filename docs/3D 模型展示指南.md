## 🔧 技术实现

### Preview.tsx 修改要点

```typescript
// 1. 混元 3D 模型 URL（从 report.json 获取）
const HUNYUAN_MODEL_URL = 'https://...cos.ap-guangzhou.tencentcos.cn/.../model.glb';

// 2. 使用 useGLTF 加载模型
function HeadModel({ url }: { url: string }) {
  const { scene } = useGLTF(url);
  return <primitive object={scene} scale={2.5} position={[0, -1, 0]} />;
}

// 3. 添加到 Canvas
<Canvas camera={{ position: [0, 0, 4], fov: 50 }}>
  <Environment preset="studio" />
  <OrbitControls autoRotate autoRotateSpeed={2} />
  <HeadModel url={modelUrl} />
  <ContactShadows position={[0, -1.5, 0]} opacity={0.4} />
</Canvas>
```
